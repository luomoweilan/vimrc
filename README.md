常用的vimrc配置

"设置desert风格
colorscheme desert

"设置行号
set nu

"设置Consolas字体风格，字体大小14
set guifont=Consolas:h14

"设置tab为4个空格键
set tabstop=4
set expandtab
set autoindent

"设置Shift+>缩进为4个空格键
set shiftwidth=4
set smartindent

"设置不保存备份文件
set nobackup

"设置搜索高亮显示
set hlsearch

"显示标尺
set ruler

"显示输入命令
set showcmd

"去掉vi一致性模式
set nocompatible
